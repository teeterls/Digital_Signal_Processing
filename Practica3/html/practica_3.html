Apartado d y e)

M = 2;
Gn = Diezmador(M,Yn);
fs_g = fs/M;
dt2 = 1/fs_g;
t_g= 0:dt2:((length(Gn)-1)/fs_g);
figure()
plot(t_g,Gn)
title('Yn diezmada con M=2')
ylabel('Gn(t)')
xlabel('Tiempo en segundos')
